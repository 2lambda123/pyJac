[docs]    def test_add_oploops(self):
        d = {'a' : [False, True]}
        op1 = optionloop(d)
        d = {'a' : [1, 2]}
        op2 = optionloop(d)
        op = op1 + op2
        for i, state in enumerate(op):
            if i < 2:
                self.assertTrue(state['a'] == i % 2)
            else:
                self.assertTrue(state['a'] == i - 1)
